template = require 'views/templates/home/business'
View = require 'views/base/view'

module.exports = class BusinessPageView extends View
  className: 'business-page'
  template: template
  autoRender: true
  autoAttach: true
  bindings: 
      '#business-name': 'name'
      '#business-country' : 
            observe: 'country_id'
            onGet: 'formatName'
      '#business-state' : 
            observe: 'state_id'
            onGet: 'formatName'
      '#business-town' : 
            oberve: 'town_id'
            onGet: 'formatName'
      '#business-zip' : 'zip_code'
      '#business-address' : 'address'
      '#business-telephone' : 'telephone'
      '#business-email' : 'email'
    events:
      'click .editable': 'handleEditableClick'
    formatName: (model) -> 
        model.get 'name' if model?
    render: ->
      super
      @stickit()

    handleEditableClick: (e) ->
      target = $ e.target
      target = target.parent() if !target.is('p')
      return if target.hasClass 'editing'
      @attr = target.data 'attr'
      @editing = target.find('span')
      @editing.hide()
      @input = $ "<input type='text' value='#{@model.get @attr}'>"
      @bindInputEvents @input
      target.addClass 'editing'
      target.append @input
      @input.focus()

    handleInputKeydown: (e) =>
      switch e.keyCode
        when 27
          console.log "Finish editing"
          @finishEditing()
        when 13
          @saveInputValue $(e.target).val()

    finishEditing: ->
      @input.off()
      @input.remove()
      @editing.parent().removeClass 'editing'
      @editing.show()

    saveInputValue: (val) ->
      @model.set @attr, val 
      console.log "Here!"
      @finishEditing()


    handleInputBlur: (e) =>
      @saveInputValue $(e.target).val()

    bindInputEvents: (input) ->
      input.on 'keydown', @handleInputKeydown
      input.on 'blur', @handleInputBlur
